DROP DATABASE AUTODB;
CREATE DATABASE AUTODB;
USE AUTODB;
CREATE TABLE AUTODB.AUTOMOTIVE(id int AUTO_INCREMENT,make VARCHAR(255), model VARCHAR(255), base_price FLOAT, PRIMARY KEY(id));
CREATE TABLE AUTODB.OPTIONSET(id INTEGER not NULL AUTO_INCREMENT, name VARCHAR(255), auto_id INTEGER not NULL, PRIMARY KEY (id), FOREIGN KEY (auto_id) REFERENCES AUTOMOTIVE(id) ON DELETE CASCADE);
CREATE TABLE AUTODB.OPTIONS(id INTEGER not NULL AUTO_INCREMENT, name VARCHAR(255), price FLOAT, optionset_id INTEGER, PRIMARY KEY (id), FOREIGN KEY(optionset_id) REFERENCES OPTIONSET(id) ON DELETE CASCADE);
